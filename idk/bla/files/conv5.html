<antArtifact identifier="cah-card-selector" type="application/vnd.ant.react" title="Cards Against Humanity Card Selector">
import React, { useState } from 'react';
import { Card, CardContent } from '@/components/ui/card';

const blackCard = "What ended my last relationship?";
const whiteCards = [
  "Pulling out",
  "Vigorous jazz hands",
  "Fingering",
  "Glenn Beck catching his scrotum on a curtain hook", 
  "The harsh light of day",
  "A homoerotic volleyball montage",
  "Spectacular abs", 
  "Genuine human connection",
  "Skeletor",
  "Darth Vader"
];

const CardSelector = () => {
  const [selectedCards, setSelectedCards] = useState([]);

  const toggleCard = (card) => {
    if (selectedCards.includes(card)) {
      setSelectedCards(selectedCards.filter(c => c !== card));
    } else if (selectedCards.length < 3) {
      setSelectedCards([...selectedCards, card]);
    }
  };

  return (
    <div className="space-y-4">
      <Card>
        <CardContent>
          <p className="text-lg font-bold mb-2">{blackCard}</p>
          <div>
            {selectedCards.map((card, index) => (
              <span key={card} className="bg-gray-200 p-1 rounded mr-1 mb-1 inline-block">
                {card}{index < selectedCards.length - 1 ? ' / ' : ''}
              </span>
            ))}
          </div>  
        </CardContent>
      </Card>

      <div className="grid grid-cols-2 gap-4">
        {whiteCards.map(card => (
          <Card 
            key={card}
            className={`cursor-pointer hover:bg-gray-100 ${
              selectedCards.includes(card) ? 'border-blue-500 border-2' : ''
            }`}
            onClick={() => toggleCard(card)}
          >
            <CardContent>{card}</CardContent>
          </Card>
        ))}
      </div>
    </div>
  );
};

export default CardSelector;
</antArtifact>
